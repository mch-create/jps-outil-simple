async function handleImport(e){ try{ const file = e.target.files[0]; if(!file) return; const data = await file.arrayBuffer(); const wb = XLSX.read(data, { type: "array" }); const ws = wb.Sheets[wb.SheetNames[0]]; // Lecture brute en tableau const aoa = XLSX.utils.sheet_to_json(ws, { header: 1, defval: "" }); if (!aoa || aoa.length < 2) { alert("Fichier vide ou structure non reconnue."); return; } rows.length = 0; // On prend les 2 premières lignes comme en-têtes possibles const header1 = aoa[0] || []; const header2 = aoa[1] || []; // Fusion des en-têtes : ex "Rémunération" + "Taux horaire" const headers = []; const maxLen = Math.max(header1.length, header2.length); for (let i = 0; i < maxLen; i++) { const h1 = String(header1[i] || "").trim(); const h2 = String(header2[i] || "").trim(); let merged = ""; if (h1 && h2) merged = h1 + " " + h2; else if (h1) merged = h1; else if (h2) merged = h2; else merged = "col_" + i; headers.push(merged); } // On démarre les données à partir de la 3e ligne const dataRows = aoa.slice(2); dataRows.forEach((row) => { if (!row || row.every(cell => String(cell || "").trim() === "")) return; const obj = {}; headers.forEach((h, i) => { obj[h] = row[i] ?? ""; }); const fonction = getVal(obj, [ "Fonction", "Emploi", "POSTE", "Qualifications", "Fonction Emploi", "Agents POSTE", "POSTE QUALIFICATIONS" ]) || "ADS"; const coef = getVal(obj, [ "Coefficient", "Coef", "Code coefficient / Indice", "Code coefficient indice", "Coefficient Coefficient", "Coef Niveau Echelon Coefficient" ]) || "130"; let heures = getVal(obj, [ "Durée mensuelle", "Duree mensuelle", "Base Mensuelle", "Base mensuelle", "Durée du travail", "Duree du travail", "Rémunération Base Mensuelle", "ETP" ]) || ""; const etp = getVal(obj, ["ETP"]) || ""; // Si ETP donné mais pas heures mensuelles if (!heures || normalize(String(heures)) === "etp") { const etpNum = toNumber(etp); if (etpNum > 0) { heures = (151.67 * etpNum).toFixed(2); } else { heures = "151.67"; } } // Taux horaire direct let brutH = getVal(obj, [ "Taux horaire", "Rémunération Taux horaire", "Salaire horaire" ]) || ""; // Salaire mensuel / base const brutM = getVal(obj, [ "Salaire de base", "Brut mensuel", "Salaire mensuel", "Base", "Rémunération Brut mensuel", "Brut mensuel Base Mensuelle" ]) || ""; // Prime ancienneté montant const primeAncienneteMontant = getVal(obj, [ "prime d'ancienneté", "prime anciennete", "Prime ancienneté", "Prime anciennete", "Prime d'ancienneté", "Prime ancienneté " ]) || 0; // Si brutH absent, on le calcule depuis le brut mensuel const hm = toNumber(heures); const bmFinal = toNumber(brutM) + toNumber(primeAncienneteMontant); if ((!brutH || toNumber(brutH) === 0) && hm > 0 && bmFinal > 0) { brutH = (bmFinal / hm).toFixed(3); } rows.push({ nom: getVal(obj, ["Nom", "NOM", "Agent", "Agents"]) || "", fonction: String(fonction).trim() || "ADS", coef: String(coef).trim() || "130", heures: String(heures).replace(",", "."), brutH: String(brutH).replace(",", "."), venteH: "", primeFixe: "0" }); }); render(); alert("Import OK : " + rows.length + " lignes"); } catch(err){ console.error(err); alert("Erreur import : " + err.message); } }
